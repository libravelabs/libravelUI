Install following dependencies:

```package-install
npm install react-aria-components lucide-react clsx twMerge
```

Add `cn` helper using `clsx` and `twMerge` for conditional class merging in Tailwind components.

```ts ts2js title="lib/utils.ts"
import { clsx, type ClassValue } from "clsx";
import { twMerge } from "tailwind-merge";

export function cn(...inputs: ClassValue[]) {
  return twMerge(clsx(inputs));
}
```

Copy and paste the following code into your project.

```tsx ts2js title="components/ui/number-field"
"use client";

import * as React from "react";
import {
  Button,
  type ButtonProps,
  NumberField as NumberFieldPrimitive,
  useLocale,
  type NumberFieldProps as NumberFieldPrimitiveProps,
  type ValidationResult,
} from "react-aria-components";
import { Minus, Plus } from "lucide-react";
import { Input } from "@/components/ui/field";
import { cn } from "@/lib/utils";

interface NumberFieldProps extends NumberFieldPrimitiveProps {
  label?: string;
  description?: string;
  placeholder?: string;
  error?: string | ((validation: ValidationResult) => string);
  indicator?: StepperButtonProps["indicator"];
}

function NumberField({
  label,
  placeholder,
  description,
  className,
  error,
  indicator,
  ...props
}: NumberFieldProps) {
  const { direction } = useLocale();

  return (
    <NumberFieldPrimitive
      {...props}
      aria-label={props["aria-label"] ?? label ?? "number-field"}
      className={cn(className, "group relative w-full")}
    >
      <Input
        label={label}
        placeholder={placeholder}
        description={description}
        error={error}
        startContent={
          <StepperButton
            indicator={indicator}
            slot="decrement"
            className={cn(
              "border-e border-border",
              direction === "ltr" ? "rounded-r-md" : "rounded-l-md"
            )}
          />
        }
        endContent={
          <StepperButton
            indicator={indicator}
            slot="increment"
            className={cn(
              "border-e border-border",
              direction === "ltr" ? "rounded-l-md" : "rounded-r-md"
            )}
          />
        }
        classNames={{ wrapper: "px-0 mx-0", input: "text-center" }}
      />
    </NumberFieldPrimitive>
  );
}

interface StepperButtonProps extends ButtonProps {
  slot: "increment" | "decrement";
  indicator?: {
    increment?: React.ReactNode;
    decrement?: React.ReactNode;
  };
  className?: string;
}

function StepperButton({
  slot,
  className,
  indicator,
  ...props
}: StepperButtonProps) {
  let icon: React.ReactNode;

  if (
    indicator &&
    typeof indicator === "object" &&
    !React.isValidElement(indicator)
  ) {
    icon = slot === "increment" ? indicator.increment : indicator.decrement;
  } else if (React.isValidElement(indicator)) {
    icon = indicator;
  } else {
    icon = slot === "increment" ? <Plus /> : <Minus />;
  }

  return (
    <Button
      className={cn(
        className,
        "size-[33px] inline-flex items-center justify-center cursor-pointer bg-muted text-muted-foreground pressed:bg-secondary pressed:text-secondary-foreground group-disabled:opacity-30 group-disabled:pointer-events-none"
      )}
      slot={slot}
      {...props}
    >
      {icon}
    </Button>
  );
}

export type { NumberFieldProps };
export { NumberField };
```

## Anatomy

```tsx
import { NumberField } from "@/components/ui/number-field";
```

```tsx
<NumberField
  label="Quantity"
  aria-label="Quantity"
  placeholder="0"
  description="Enter the desired amount."
  error="Quantity is required"
  className="w-full max-w-sm"
  defaultValue={1}
  minValue={1}
  maxValue={10}
  step={1}
  formatOptions={{ style: "decimal" }}
  onChange={(value) => console.log("Changed:", value)}
  indicator={{
    increment: <ChevronUp className="w-4 h-4" />,
    decrement: <ChevronDown className="w-4 h-4" />
  }}
/>
```
