## Installation

<Tabs items={["Manual", "CLI"]} defaultIndex={1}>
<Tab value="Manual">

Install following dependencies:

```package-install
npm install react-aria-components clsx tailwind-merge
```

Add `cn` helper using `clsx` and `twMerge` for conditional class merging in Tailwind components.

```ts ts2js title="lib/utils.ts"
import { clsx, type ClassValue } from "clsx";
import { twMerge } from "tailwind-merge";

export function cn(...inputs: ClassValue[]) {
  return twMerge(clsx(inputs));
}
```

Copy and paste the following code into your project.

```tsx ts2js title="components/ui/separator"
import {
  Separator as Divider,
  type SeparatorProps,
} from "react-aria-components";
import { cn } from "@/lib/utils";

interface CustomSeparatorProps extends SeparatorProps {
  text?: string;
}

function Separator({
  orientation = "horizontal",
  text,
  className,
  ...props
}: CustomSeparatorProps) {
  const isHorizontal = orientation === "horizontal";

  if (!text) {
    return (
      <Divider
        orientation={orientation}
        {...props}
        className={cn(
          "bg-border shrink-0",
          isHorizontal ? "h-px w-full" : "w-px h-full",
          className
        )}
      />
    );
  }

  return (
    <div
      role="separator"
      aria-orientation={orientation}
      className={cn(
        "flex items-center justify-center",
        isHorizontal ? "flex-row w-full" : "flex-col h-full",
        className
      )}
    >
      <Divider
        orientation={orientation}
        {...props}
        className={cn(
          "bg-border shrink-0",
          isHorizontal ? "h-px flex-1" : "w-px flex-1"
        )}
      />
      <span
        className={cn(
          "text-sm text-muted-foreground bg-background",
          isHorizontal ? "px-2" : "py-2"
        )}
      >
        {text}
      </span>
      <Divider
        orientation={orientation}
        {...props}
        className={cn(
          "bg-border shrink-0",
          isHorizontal ? "h-px flex-1" : "w-px flex-1"
        )}
      />
    </div>
  );
}

export { Separator };
```
