## Anatomy

```tsx
import { Stepper } from "@/components/ui/core/stepper";

import {
  StepperRoot,
  StepperItem,
  StepperTrigger,
  StepperIndicator,
  StepperTitle,
  StepperDescription,
  StepperSeparator,
  StepperContent,
  StepperItems,
} from "@/components/ui/core/stepper";
```

```tsx
<Stepper steps={steps} activeStep={step} onStepChange={setStep} />

/* Or */

<StepperRoot>
  <StepperItems>
    <StepperItem>
      <StepperTrigger>
        <StepperIndicator />
        <div>
          <StepperTitle>{title}</StepperTitle>
          <StepperDescription>{description}</StepperDescription>
        </div>
      </StepperTrigger>

      <StepperSeparator />
    </StepperItem>
  </StepperItems>
</StepperRoot>
```
