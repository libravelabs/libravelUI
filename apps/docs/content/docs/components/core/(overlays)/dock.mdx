## Anatomy

```tsx
import {
  DockContent,
  DockHeader,
  DockTitle,
  DockDescription,
  Dock,
  DockFooter,
  DockTrigger,
  DockBody,
} from "@/components/ui/core/dock";
```

```tsx
<Dock>
  <DockTrigger>Open Dock</DockTrigger>

  <DockContent side="right">
    <DockHeader>

    </DockHeader>
```
