## Anatomy

<Info tone="warning" title="Scale Background" icon={<TriangleAlert />}>
  **Note:** If you use `shouldScaleBackground`, add `data-dock` to the parent
  container (e.g. your main layout wrapper) so it can be scaled correctly.
</Info>

```tsx
import {
  DockContent,
  DockHeader,
  DockTitle,
  DockDescription,
  Dock,
  DockFooter,
  DockTrigger,
  DockBody,
} from "@/components/ui/core/dock";
```

```tsx
<Dock>
  <DockTrigger>Open Dock</DockTrigger>

  <DockContent side="right" shouldScaleBackground>
    <DockHeader>
      <DockTitle>CONTENT BELONGS HERE</DockTitle>
      <DockDescription>CONTENT BELONGS HERE</DockDescription>
    </DockHeader>

    <DockBody>CONTENT BELONGS HERE</DockBody>

    <DockFooter>CONTENT BELONGS HERE</DockFooter>
  </DockContent>
</Dock>
```
