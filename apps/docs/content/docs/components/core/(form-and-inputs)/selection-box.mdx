## Anatomy

```tsx
import {
  Selectionbox,
  SelectionboxItem,
} from "@/components/ui/core/selection-box";
```

```tsx
<Selectionbox selectionMode="multiple">
  <SelectionboxItem
    label="Label belongs here"
    description="Some description."
  />
</Selectionbox>
```
